#include <stdio.h>
//10. Dois alienígenas crescem indefinidamente. Zerg tem 1,50m e cresce 2cm por ano enquanto Splim tem 1,30m e cresce 3cm por ano.
// Faça um programa que calcule quantos anos serão necessários para que Splim seja maior que Zerg.

#include <stdio.h>

int main() {
    // Alturas iniciais em metros
    float zerg = 1.50, splin = 1.30;
    // Taxas de crescimento em metros por ano
    float crescimento_zerg = 0.02, crescimento_splin = 0.03;
    // Contador de anos
    int anos = 0;

    // Enquanto Splim for menor que Zerg
    while (splin < zerg) {
        // Incrementa as alturas
        zerg += crescimento_zerg;
        splin += crescimento_splin;
        // Incrementa o contador de anos
        anos++;
    }

    printf("Levara %d anos para que Splin seja maior que Zerg.\n", anos);

    return 0;
}
